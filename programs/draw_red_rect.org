MAIN:
    MOV R0, 0x8000              ; i = R0
    MOV R1, 0x1                 ; i += 1
    MOV R2, 0xC000              ; i < R2
    MOV R3, 0xF800              ; Color
    JMP DRAWING_LOOP

DRAWING_LOOP:
    STORER R3, R0
    ADD R0, R0, R1              ; R0 += R1 --> R0 += 1
    CMP R0, R2
    JNE DRAWING_LOOP
    JE END


END:
    HLT
